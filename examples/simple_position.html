def run_sim(env, wps):
    observations = collections.deque()
    obs, info = env.reset(seed=1337)

    for setpoint in tqdm(wps):
        obs, reward, term, trunc, info = env.step(setpoint)

        observations.append(np.copy(obs))

        if trunc:
            print(f"{trunc=} {term=} {info=}")
            break

    return observations
